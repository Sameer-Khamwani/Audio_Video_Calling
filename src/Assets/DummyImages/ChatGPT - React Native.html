To achieve a custom scrollbar on the right side with a blue-filled bar that reflects the scroll position, we need to:

import React, { useRef, useState } from 'react';
import {
  ScrollView,
  View,
  StyleSheet,
  Animated,
  Dimensions,
} from 'react-native';
import AppWrapper from '../../Components/AppWrapper';
import CustomText from '../../Components/CustomText';
import { font, heightPixel } from '../../Utils/Dimensions';
import Theme from '../../Utils/Theme';

const { height: screenHeight } = Dimensions.get('window');

const ContentScreen = () => {
  const scrollY = useRef(new Animated.Value(0)).current;
  const [contentHeight, setContentHeight] = useState(1);

  const handleContentSizeChange = (_, contentHeight) => {
    setContentHeight(contentHeight);
  };

  const scrollIndicatorHeight = scrollY.interpolate({
    inputRange: [0, Math.max(1, contentHeight - screenHeight)],
    outputRange: [heightPixel(40), heightPixel(100)], // Adjusted scroll bar height range
    extrapolate: 'clamp',
  });

  const scrollIndicatorTranslateY = scrollY.interpolate({
    inputRange: [0, Math.max(1, contentHeight - screenHeight)],
    outputRange: [0, screenHeight - heightPixel(140)], // Ensures it moves within bounds
    extrapolate: 'clamp',
  });

  return (
    <AppWrapper>
      <View style={styles.container}>
        <ScrollView
          showsVerticalScrollIndicator={false}
          onContentSizeChange={handleContentSizeChange}
          onScroll={Animated.event(
            [{ nativeEvent: { contentOffset: { y: scrollY } } }],
            { useNativeDriver: false }
          )}>
          <CustomText
            size={font(16)}
            weight="semibold"
            text={'Lorem Ipsum'}
            style={styles.title}
          />
          <CustomText size={font(12)} text={loremIpsumText} />
        </ScrollView>

        {/* Custom Scroll Bar */}
        <View style={styles.scrollBar}>
          <Animated.View
            style={[styles.scrollThumb, { height: scrollIndicatorHeight, transform: [{ translateY: scrollIndicatorTranslateY }] }]}
          />
        </View>
      </View>
    </AppWrapper>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexDirection: 'row',
  },
  title: {
    marginBottom: heightPixel(6),
  },
  scrollBar: {
    position: 'absolute',
    right: 5,
    top: 10,
    width: 6,
    height: '90%',
    backgroundColor: Theme.lightText, // Light gray background for scrollbar
    borderRadius: 10,
  },
  scrollThumb: {
    width: '100%',
    backgroundColor: Theme.primaryBlue, // Blue scroll indicator
    borderRadius: 10,
  },
});

const loremIpsumText = `Lorem Ipsum is simply dummy text of the printing and typesetting industry...`;
// Full text truncated for readability, keep your original content here

export default ContentScreen;
